// Function to fetch the lockdown status
function fetchLockdownStatus() {
    fetch('lockdown-status.txt')
        .then(response => {
            if (!response.ok) {
                throw new Error('Lockdown status file not found');
            }
            return response.text();
        })
        .then(status => {
            if (status.trim() === 'enabled') {
                // Lockdown is enabled, perform actions accordingly
                console.log('Lockdown is enabled');
            } else {
                // Lockdown is disabled
                console.log('Lockdown is disabled');
            }
        })
        .catch(error => {
            console.error('Error fetching lockdown status:', error.message);
        });
}

// Call the function to fetch lockdown status when the page loads
document.addEventListener('DOMContentLoaded', fetchLockdownStatus);
